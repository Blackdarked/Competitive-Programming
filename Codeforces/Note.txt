int n; cin >> n;
    vi a(n), b(n);
    rep (i, 0, n) {
        cin >> a[i] >> b[i];
    }
    int l = 0, h = n;
    while (l < h) {
        int m = (1 + l + h) >> 1, j = 0;
        bool c = 1;
        rep (i, 1, m+1) {
            while (j < n and (a[j] < m - i or b[j] < i - 1)) {
                ++j;
            }
            if (j == n) {
                c = 0
                break;
            }
            ++j;
        }
        c ? l = m : h = m - 1;
    }
    cout << l << '\n';
}
